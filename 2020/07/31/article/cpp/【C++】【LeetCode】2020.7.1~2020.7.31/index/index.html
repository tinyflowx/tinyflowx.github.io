32. Longest Valid Parentheses

[cpp] class Solution { public: //()() //()(()() //())() //((())) int longestValidParentheses(string s) { int maxLen = 0; vector dp(s.size(), 0); for (int i = 0; i < s.size(); ++i) { if (s[i] == ‘(‘ || (i == 0 && s[i] == ‘)’)) { continue; } //dp[i]代表当前i位置的最大匹配数 int matchIndex = i - dp[i - 1] - 1; if (matchIndex < 0 || s[matchIndex] == ‘)’) { continue; } //如果i - dp[i - 1] - 1是左括号的话，dp[i] = dp[i - 1] + 2 dp[i] = dp[i - 1] + 2; //考虑()()的情况dp[i]还要加上上一段匹配长度 if (i - dp[i] > 0) { dp[i] += dp[i - dp[i]]; } maxLen = dp[i] > maxLen ? dp[i] : maxLen; }

return maxLen; } }; [/cpp]

42. Trapping Rain Water
